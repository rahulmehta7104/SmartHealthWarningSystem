SELECT c.city_name, d.disease_name, COUNT(*) as cases FROM patient_reports pr JOIN cities c ON pr.city_id = c.city_id JOIN diseases d ON pr.disease_id = d.disease_id GROUP BY c.city_id, d.disease_id ORDER BY cases DESC LIMIT 10;
Top disease cases by city
SELECT city_name, AVG(air_quality_index) as avg_aqi FROM environmental_data ed JOIN cities c ON ed.city_id = c.city_id GROUP BY c.city_id ORDER BY avg_aqi DESC LIMIT 10;
Cities with highest air pollution
SELECT h.hospital_name, c.city_name, h.bed_capacity, COUNT(pr.report_id) as patients FROM hospitals h JOIN cities c ON h.city_id = c.city_id LEFT JOIN patient_reports pr ON h.hospital_id = pr.hospital_id GROUP BY h.hospital_id ORDER BY patients DESC LIMIT 10;
